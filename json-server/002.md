import { useState, useEffect } from 'react';
import { useNavigate, useParams } from 'react-router-dom';
import { createPost, getPostById, updatePost } from '../api/posts';

const PostForm = () => {
  const navigate = useNavigate();
  const { id } = useParams(); // ìˆ˜ì •ì´ë©´ id ì¡´ì¬
  const isEdit = !!id;

  const [form, setForm] = useState({
    title: '',
    content: '',
    author: '',
  });

  useEffect(() => {
    if (isEdit) {
      getPostById(id).then((res) => setForm(res.data));
    }
  }, [id]);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm((prev) => ({ ...prev, [name]: value }));
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (isEdit) {
      await updatePost(id, form);
    } else {
      await createPost({
        ...form,
        createdAt: new Date().toISOString(),
      });
    }
    navigate('/');
  };

  return (
    <div>
      <h2>{isEdit ? 'ê²Œì‹œê¸€ ìˆ˜ì •' : 'ê²Œì‹œê¸€ ì‘ì„±'}</h2>
      <form onSubmit={handleSubmit}>
        <input
          name="title"
          placeholder="ì œëª©"
          value={form.title}
          onChange={handleChange}
        /><br />
        <textarea
          name="content"
          placeholder="ë‚´ìš©"
          value={form.content}
          onChange={handleChange}
        /><br />
        <input
          name="author"
          placeholder="ì‘ì„±ì"
          value={form.author}
          onChange={handleChange}
        /><br />
        <button type="submit">{isEdit ? 'ìˆ˜ì •' : 'ë“±ë¡'}</button>
