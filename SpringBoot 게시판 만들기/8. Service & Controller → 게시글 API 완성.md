# 10) 서비스 계층

`src/main/java/com/example/boardapi/service/PostService.java`

```java
package com.example.boardapi.service;

import com.example.boardapi.domain.Post;
import com.example.boardapi.dto.PageResponse;
import com.example.boardapi.dto.PostCreateRequest;
import com.example.boardapi.dto.PostResponse;
import com.example.boardapi.dto.PostUpdateRequest;
import com.example.boardapi.exception.ApiException;
import com.example.boardapi.exception.ErrorCode;
import com.example.boardapi.mapper.PostMapper;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

@Service
public class PostService {

    private final PostMapper postMapper;

    public PostService(PostMapper postMapper) {
        this.postMapper = postMapper;
    }

    @Transactional
    public Long create(PostCreateRequest req) {
        Post p = new Post();
        p.setTitle(req.title());
        p.setContent(req.content());
        p.setAuthorId(req.authorId());
        postMapper.insert(p);
        return p.getId();
    }

    @Transactional
    public void update(Long id, PostUpdateRequest req) {
        Post found = postMapper.findById(id);
        if (found == null) throw new ApiException(ErrorCode.NOT_FOUND);
        found.setTitle(req.title());
        found.setContent(req.content());
        postMapper.update(found);
    }

    @Transactional
    public void delete(Long id) {
        Post found = postMapper.findById(id);
        if (found == null) throw new ApiException(ErrorCode.NOT_FOUND);
        postMapper.deleteById(id);
    }

    @Transactional
    public PostResponse findOne(Long id, boolean increaseView) {
        Post found = postMapper.findById(id);
        if (found == null) throw new ApiException(ErrorCode.NOT_FOUND);
        if (increaseView) postMapper.increaseViews(id);
        Post after = postMapper.findById(id);
        return toResponse(after);
    }

    public PageResponse<PostResponse> findAll(int page, int size, String keyword) {
        int current = Math.max(1, page);
        int limit = Math.max(1, size);
        int offset = (current - 1) * limit;

        List<Post> rows = postMapper.findAll(offset, limit, keyword);
        long total = postMapper.countAll(keyword);
        List<PostResponse> items = rows.stream().map(this::toResponse).toList();

        return PageResponse.of(items, current, limit, total);
    }

    private PostResponse toResponse(Post p) {
        return new PostResponse(
                p.getId(), p.getTitle(), p.getContent(),
                p.getAuthorId(), p.getViews(),
                p.getCreatedAt(), p.getUpdatedAt()
        );
    }
}
```

# 11) 컨트롤러 (REST API)

`src/main/java/com/example/boardapi/controller/PostController.java`

```java
package com.example.boardapi.controller;

import com.example.boardapi.dto.PageResponse;
import com.example.boardapi.dto.PostCreateRequest;
import com.example.boardapi.dto.PostResponse;
import com.example.boardapi.dto.PostUpdateRequest;
import com.example.boardapi.service.PostService;
import com.example.boardapi.util.ApiResponse;
import jakarta.validation.Valid;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/posts")
public class PostController {

    private final PostService postService;
    public PostController(PostService postService){ this.postService = postService; }

    // 목록/검색/페이징
    @GetMapping
    public ApiResponse<PageResponse<PostResponse>> list(
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "10") int size,
            @RequestParam(required = false) String keyword
    ) {
        return ApiResponse.ok(postService.findAll(page, size, keyword));
    }

    // 상세 (조회수 증가)
    @GetMapping("/{id}")
    public ApiResponse<PostResponse> detail(@PathVariable Long id) {
        return ApiResponse.ok(postService.findOne(id, true));
    }

    // 등록
    @PostMapping
    public ApiResponse<Long> create(@RequestBody @Valid PostCreateRequest req) {
        return ApiResponse.ok(postService.create(req));
    }

    // 수정
    @PutMapping("/{id}")
    public ApiResponse<Void> update(@PathVariable Long id, @RequestBody @Valid PostUpdateRequest req) {
        postService.update(id, req);
        return ApiResponse.ok(null, "updated");
    }

    // 삭제
    @DeleteMapping("/{id}")
    public ApiResponse<Void> delete(@PathVariable Long id) {
        postService.delete(id);
        return ApiResponse.ok(null, "deleted");
    }
}
```
